CREATE TABLE [dbo].[CARGOS] (
    [ID_CARGO] INT          IDENTITY (1, 1) NOT NULL,
    [NOMBRE]   VARCHAR (20) NOT NULL,
    CONSTRAINT [PK_CARGOS] PRIMARY KEY CLUSTERED ([ID_CARGO] ASC)
);

CREATE TABLE [dbo].[USUARIOS] (
    [ID_USUARIO] INT          IDENTITY (1, 1) NOT NULL,
    [NOMBRE]     VARCHAR (20) NOT NULL,
    [APELLIDO]   VARCHAR (20) NOT NULL,
    [USUARIO]    VARCHAR (12) NOT NULL,
    [CLAVE]      VARCHAR (12) NOT NULL,
    [ID_CARGO]   INT NOT NULL,
    [DIRECCION]  VARCHAR (150) NOT NULL,
    [TELEFONO]   VARCHAR (15) NOT NULL,
    CONSTRAINT [PK_USUARIOS] PRIMARY KEY CLUSTERED ([ID_USUARIO] ASC),
    CONSTRAINT [FK_USUARIOS_CARGOS] FOREIGN KEY ([ID_CARGO]) REFERENCES [dbo].[CARGOS] ([ID_CARGO])
);

CREATE TABLE [dbo].[CLIENTES] (
    [ID_CLIENTE] INT          IDENTITY (1, 1) NOT NULL,
    [NOMBRE]     VARCHAR (20) NOT NULL,
    [APELLIDO]   VARCHAR (20) NOT NULL,
    [DIRECCION]  VARCHAR(150) NOT NULL,
    [TELEFONO]   VARCHAR(15)  NOT NULL,
    [IDENTIDAD]  VARCHAR(15)  NOT NULL,
    [FECHANAC]   DATETIME     NOT NULL,
    CONSTRAINT [PK_CLIENTES] PRIMARY KEY CLUSTERED ([ID_CLIENTE] ASC)
);

CREATE TABLE [dbo].[AVIONES] (
    [ID_AVION]    INT          IDENTITY (1, 1) NOT NULL,
    [CAPACIDAD]   INT 	       NOT NULL,
    [LONGITUD]    VARCHAR (12) NOT NULL,
    [ENVERGADURA] VARCHAR (12) NOT NULL,
    [VELOCIDAD]   VARCHAR (12) NOT NULL,
    [COMPANIA]    VARCHAR (12) NOT NULL,
    [NUMERO]      INT	       NOT NULL,
    [FECHADQUISICION] DATETIME NOT NULL,
    CONSTRAINT [PK_AVIONES] PRIMARY KEY CLUSTERED ([ID_AVION] ASC)
);

CREATE TABLE [dbo].[VUELOS] (
    [ID_VUELO]       INT IDENTITY (1, 1) NOT NULL,
    [HORARIOSALIDA]  DATETIME NOT NULL,
    [HORARIOLLEGADA] DATETIME NOT NULL,
    [ORIGEN]         VARCHAR(15) NOT NULL,
    [DESTINO]        VARCHAR(15) NOT NULL,
    [ID_AVION]       INT	 NOT NULL,
    [ID_PILOTO]      INT	 NOT NULL,
    CONSTRAINT [PK_VUELOS] PRIMARY KEY CLUSTERED ([ID_VUELO] ASC),
    CONSTRAINT [FK_VUELOS_AVIONES] FOREIGN KEY ([ID_AVION]) REFERENCES [dbo].[AVIONES] ([ID_AVION]),
    CONSTRAINT [FK_VUELOS_USUARIOS] FOREIGN KEY ([ID_PILOTO]) REFERENCES [dbo].[USUARIOS] ([ID_USUARIO])
);

CREATE TABLE [dbo].[RESERVACIONES] (
    [ID_RESERVACION] INT IDENTITY (1, 1) NOT NULL,
    [ID_CLIENTE]     INT NOT NULL,
    [ID_VUELO]       INT NOT NULL,
    [NUMERO_ASIENTO] INT NOT NULL,
    [FECHA]          DATETIME NOT NULL,
    [PRECIO]         DECIMAL NOT NULL,
    [ISV]            DECIMAL NOT NULL,
    [DESCUENTO]      DECIMAL NOT NULL,
    [TOTAL]          DECIMAL NOT NULL,
    CONSTRAINT [PK_RESERVACIONES] PRIMARY KEY CLUSTERED ([ID_RESERVACION] ASC),
    CONSTRAINT [FK_RESERVACIONES_CLIENTES] FOREIGN KEY ([ID_CLIENTE]) REFERENCES [dbo].[CLIENTES] ([ID_CLIENTE]),
    CONSTRAINT [FK_RESERVACIONES_VUELOS] FOREIGN KEY ([ID_VUELO]) REFERENCES [dbo].[VUELOS] ([ID_VUELO])
);
